//c++的输入输出 
#include <iostream>
using namespace std;
int main(){
	int c;
	char b[10];
	cin >> c ;//若不事先声明命名空间，可以用std::cin/cout 
	cin.ignore (100,'n');//忽略100字节输入，除非遇到n //n也被拦截 ,
	cin >> b;
	if(cin.eof()){
		cout << "到达文件末尾";perror("");
		exit(0);}
	if(cin.fail()) {
		cout << "cin无法工作";perror("");
		exit(0);}
	if(cin.bad()){
		cout << "出现严重问题比如内存不够";perror("");
		exit(0);}
	if(cin.good())
		cout << "输入成功"; 
	cout << "1234" << c << b << endl;//endl表示'\n' 
	cout << "请输入任意一个字符结束程序";
	cin.ignore(100,'\n');//可以清除缓冲区 
	cin.get(); //输入单个字符，暂停程序 
}
