/*5:整数奇偶排序
描述
输入10个整数，彼此以空格分隔
重新排序以后输出(也按空格分隔)，要求:
1.先输出其中的奇数,并按从大到小排列；
2.然后输出其中的偶数,并按从小到大排列。
输入
任意排序的10个整数（0～100），彼此以空格分隔
输出
按照要求排序后输出，由空格分隔
样例输入
4 7 3 13 11 12 0 47 34 98
样例输出
47 13 11 7 3 0 4 12 34 98
提示
1. 测试数据可能有很多组，请使用while(cin>>a[0]>>a[1]>>...>>a[9])类似的做法来实现；
2. 输入数据随机，有可能相等。*/
#include<stdio.h>
int main(){
	int a[10];
    int b[10];
	int c[10];
	int i,j,k,m,n;
	while(scanf("%d",&a[0])!=EOF){
	
	for(i=1,j=0,k=0;i<10;i++){
		scanf("%d",&a[i]);
		if(a[i]%2==1)b[j++]=a[i];
		else c[k++]=a[i];
	} 
	if(a[0]%2==1)b[j++]=a[0];
		else c[k++]=a[0];
    i=0;
    int temp; 
    for(i=1;i<j;i++){//从前向后取数 
    	temp=b[i];
    	m=i-1;
    	while(temp>b[m]&&m>-1){//比较，找到位置 
    		b[m+1]=b[m];
    		m--;
    	}
    	if(m!=i-1) b[m+1]=temp;//temp>a[i-1],不用插到前面 
    }
    i=0;

    for(i=1;i<k;i++){//从前向后取数 
    	temp=c[i];
    	m=i-1;
    	while(temp<c[m]&&m>-1){//比较，找到位置 
    		c[m+1]=c[m];
    		m--;
    	}
    	if(m!=i-1) c[m+1]=temp;//temp>a[i-1],不用插到前面 
    }
    int flag=0;
    for(i=0;i<j;i++){
    	if(flag>0)printf(" ");
    	printf("%d",b[i]);flag=1;
    }
    for(i=0;i<k;i++){if(flag>0)printf(" ");
    printf("%d",c[i]);flag=1;
}printf("\n");
}}
