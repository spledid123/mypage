#include<stdio.h>

#define pi 3.14//宏定义就是简单的替换 
#define s pi*2*2//宏定义的嵌套(一定在下面）
#define t(x,y) x*y//可以替换为一些简单函数，但是功能有限而且替代不了函数 
#define To( x, y ) x##y//内联函数，简单替换可以做到神奇地事情 
int main(){
	printf("pi=%f,s=%f\nt(1+2,4)=%d,t((1,2),4)=%d\nTo(2,50)=%d",pi,s,t(1+2,4),t((1,2),4),To(2,50));
	//pi=3.140000,s=12.560000
	//t(1+2,4)=9,t((1,2),4)=8
	//To(2,50)=250
}
